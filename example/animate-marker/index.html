Animate a marker

Animate the position of a Marker by updating its location on each frame.

<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8" />
<title>Animate a marker</title>
<meta name="viewport" content="initial-scale=1,maximum-scale=1,user-scalable=no" />
<script src="https://unpkg.com/maplibre-gl@3.0.0-pre.1/dist/maplibre-gl.js"></script>
<link href="https://unpkg.com/maplibre-gl@3.0.0-pre.1/dist/maplibre-gl.css" rel="stylesheet" />
<style>
	body { margin: 0; padding: 0; }
	#map { position: absolute; top: 0; bottom: 0; width: 100%; }
</style>
</head>
<body>
<div id="map"></div>
<script>
var map = new maplibregl.Map({
container: 'map',
style: 'https://api.maptiler.com/maps/streets/style.json?key=get_your_own_OpIi9ZULNHzrESv6T2vL',
center: [0, 0],
zoom: 2
});
 
var marker = new maplibregl.Marker();
 
function animateMarker(timestamp) {
var radius = 20;
 
// Update the data to a new position based on the animation timestamp. The
// divisor in the expression `timestamp / 1000` controls the animation speed.
marker.setLngLat([
Math.cos(timestamp / 1000) * radius,
Math.sin(timestamp / 1000) * radius
]);
 
// Ensure it's added to the map. This is safe to call if it's already added.
marker.addTo(map);
 
// Request the next frame of the animation.
requestAnimationFrame(animateMarker);
}
 
// Start the animation.
requestAnimationFrame(animateMarker);
</script>
 
</body>
</html>
